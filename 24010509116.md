⚙️ 2. Temel Kullanım Talimatları
Uygulama arayüzü tek bir ekrandan oluşur ve günlük rutinlerinizi yönetmenizi sağlayan 5 ana bölüm barındırır:

🧑‍াগ Profil ve Hedef Ayarları (İlk Kullanım)
Sağ üst köşede bulunan Ayar / Profil İkonuna (dişli çarklı hesap butonu) dokunun.

Açılan pencerede İsminizi ve Günlük Su Hedefinizi (Bardak cinsinden) girerek "Kaydet ve Kapat" butonuna basın. Uygulama artık size isminizle hitap edecektir.

💧 Su Takibi Paneli
Gün içinde su içtikçe "1 Bardak Ekle" butonuna basarak sayacı artırın.

İçtiğiniz miktar, belirlediğiniz günlük hedefe ulaştığında uygulama ekranda bir kutlama tebriki (Tebrikler! 🎉) gösterecektir.

Yeni bir güne başlarken sağ üstteki "Sıfırla" yazısına dokunarak su sayacını sıfırlayabilirsiniz.

😊 Mod Durumu Belirleme
"Bugün Nasıl Hissediyorsun?" başlığı altında yer alan mod butonlarından (Harika, Normal, Yorgun, Stresli) durumunuza uygun olanına dokunun. Seçtiğiniz mod renk değiştirerek aktif hale gelecektir.

⏱️ Egzersiz Süresi Kronometresi
Antrenmana veya aktiviteye başlarken "Başlat" butonuna basın.

Dinlenme aralarında veya egzersiz bittiğinde "Duraklat" yapabilirsiniz.

Süreyi tamamen sıfırlamak için "Sıfırla" butonunu kullanabilirsiniz.

🏃‍♂️ BMI (Vücut Kitle Endeksi) ve Egzersiz Önerisi
Boy (cm) ve Kilo (kg) alanlarına güncel bilgilerinizi yazın.

"Hesapla ve Öneri Al" butonuna dokunun.

Sistem anında vücut kitle endeksinizi hesaplayacak, Zayıf, Normal, Fazla Kilolu veya Obez kategorilerinden hangisinde olduğunuzu belirtecek ve durumunuza özel sağlıklı egzersiz tavsiyesini alt panelde listeleyecektir.

📝 Günlük Notlarım
"Notunuzu yazın..." alanına dokunarak o günkü spor, beslenme veya ruh halinizle ilgili notlar alın ve "Ekle" butonuna basın.

Notlarınız, eklendiği saat bilgisiyle birlikte alt alta listelenir.

Silmek istediğiniz notun sağ tarafında bulunan Kırmızı Çöp Kutusu ikonuna dokunarak notu kaldırabilirsiniz.
import React, { useState, useEffect } from 'react';
import { 
  StyleSheet, 
  Text, 
  View, 
  TextInput, 
  TouchableOpacity, 
  ScrollView, 
  Alert, 
  Modal 
} from 'react-native';
import AsyncStorage from '@react-native-async-storage/async-storage';
import { MaterialCommunityIcons } from '@expo/vector-icons';

export default function App() {
  // --- STATE TANIMLAMALARI ---
  const [waterCount, setWaterCount] = useState(0);
  const [mood, setMood] = useState('');
  const [weight, setWeight] = useState('');
  const [height, setHeight] = useState('');
  const [bmiResult, setBmiResult] = useState(null);
  const [bmiCategory, setBmiCategory] = useState('');
  const [exerciseRecommendation, setExerciseRecommendation] = useState('');
  
  // Kronometre State'leri
  const [timer, setTimer] = useState(0);
  const [isTimerActive, setIsTimerActive] = useState(false);
  
  // Günlük Notlar State'leri
  const [notes, setNotes] = useState([]);
  const [currentNote, setCurrentNote] = useState('');
  
  // Profil/Hedef Modalı State'leri
  const [isProfileModalVisible, setIsProfileModalVisible] = useState(false);
  const [userName, setUserName] = useState('');
  const [dailyWaterGoal, setDailyWaterGoal] = useState('8'); // Varsayılan 8 bardak

  // --- KRONOMETRE LOGIC ---
  useEffect(() => {
    let interval = null;
    if (isTimerActive) {
      interval = setInterval(() => {
        setTimer((prevTime) => prevTime + 1);
      }, 1000);
    } else if (!isTimerActive && timer !== 0) {
      clearInterval(interval);
    }
    return () => clearInterval(interval);
  }, [isTimerActive, timer]);

  const formatTime = (timeInSeconds) => {
    const minutes = Math.floor(timeInSeconds / 60);
    const seconds = timeInSeconds % 60;
    return `${minutes < 10 ? '0' : ''}${minutes}:${seconds < 10 ? '0' : ''}${seconds}`;
  };

  const handleStartPauseTimer = () => {
    setIsTimerActive(!isTimerActive);
  };

  const handleResetTimer = () => {
    setIsTimerActive(false);
    setTimer(0);
  };

  // --- SU TAKİBİ LOGIC ---
  const incrementWater = () => {
    const goal = parseInt(dailyWaterGoal) || 8;
    if (waterCount + 1 >= goal) {
      Alert.alert("Tebrikler!", "Günlük su tüketimi hedefinize ulaştınız! 🎉");
    }
    setWaterCount(waterCount + 1);
  };

  const resetWater = () => {
    setWaterCount(0);
  };

  // --- BMI (VÜCUT KİTLE ENDEKSİ) VE EGZERSİZ LOGIC ---
  const calculateBMI = () => {
    const w = parseFloat(weight);
    const h = parseFloat(height) / 100; // cm'yi metreye çevir

    if (!w || !h || h <= 0 || w <= 0) {
      Alert.alert("Hata", "Lütfen geçerli boy ve kilo değerleri giriniz.");
      return;
    }

    const bmi = (w / (h * h)).toFixed(1);
    setBmiResult(bmi);

    let category = '';
    let recommendation = '';

    if (bmi < 18.5) {
      category = 'Zayıf';
      recommendation = 'Hafif tempolu yürüyüşler ve kas kütlesini artırmaya yönelik hafif direnç egzersizleri yapabilirsiniz. Beslenmenize dikkat etmeniz önerilir.';
    } else if (bmi >= 18.5 && bmi < 24.9) {
      category = 'Normal';
      recommendation = 'Harika! Formunuzu korumak için haftada 3-4 gün 30 dakikalık kardiyo (koşu, bisiklet) ve tüm vücut egzersizleri idealdir.';
    } else if (bmi >= 24.9 && bmi < 29.9) {
      category = 'Fazla Kilolu';
      recommendation = 'Yağ yakımını desteklemek için yüksek tempolu yürüyüşler, yüzme, hafif koşular veya HIIT (Yüksek Yoğunluklu Antrenman) programları uygulayabilirsiniz.';
    } else {
      category = 'Obez';
      recommendation = 'Eklemlerinize aşırı yük bindirmemek adına düşük tempolu uzun yürüyüşler, tempolu yüzme ve uzman kontrolünde hafif egzersizler başlangıç için en güvenlisidir.';
    }

    setBmiCategory(category);
    setExerciseRecommendation(recommendation);
  };

  // --- GÜNLÜK NOTLAR LOGIC ---
  const addNote = () => {
    if (currentNote.trim() === '') {
      Alert.alert("Hata", "Boş bir not ekleyemezsiniz.");
      return;
    }
    const newNote = {
      id: Date.now().toString(),
      text: currentNote,
      date: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' })
    };
    setNotes([newNote, ...notes]);
    setCurrentNote('');
  };

  const deleteNote = (id) => {
    setNotes(notes.filter(note => note.id !== id));
  };

  // --- RENDERING ---
  return (
    <ScrollView style={styles.container} contentContainerStyle={{ paddingBottom: 40 }}>
      {/* BAŞLIK VE PROFİL BUTONU */}
      <View style={styles.header}>
        <View>
          <Text style={styles.welcomeText}>Merhaba, {userName || 'Kullanıcı'} 👋</Text>
          <Text style={styles.appTitle}>Loop Daily</Text>
        </View>
        <TouchableOpacity 
          style={styles.profileButton} 
          onPress={() => setIsProfileModalVisible(true)}
        >
          <MaterialCommunityIcons name="account-cog" size={28} color="#fff" />
        </TouchableOpacity>
      </View>

      {/* MOD DURUMU PANELİ */}
      <View style={styles.card}>
        <Text style={styles.cardTitle}>Bugün Nasıl Hissediyorsun?</Text>
        <View style={styles.moodContainer}>
          {['😊 Harika', '😐 Normal', '😔 Yorgun', '😡 Stresli'].map((m) => (
            <TouchableOpacity
              key={m}
              style={[styles.moodButton, mood === m && styles.moodButtonSelected]}
              onPress={() => setMood(m)}
            >
              <Text style={[styles.moodButtonText, mood === m && styles.moodButtonTextSelected]}>{m}</Text>
            </TouchableOpacity>
          ))}
        </View>
      </View>

      {/* SU TAKİP PANELİ */}
      <View style={styles.card}>
        <View style={styles.cardHeaderRow}>
          <Text style={styles.cardTitle}>Su Takibi 💧</Text>
          <TouchableOpacity onPress={resetWater}>
            <Text style={styles.resetText}>Sıfırla</Text>
          </TouchableOpacity>
        </View>
        <Text style={styles.waterSubtitle}>Bugün içilen su miktarı:</Text>
        <View style={styles.waterDisplayRow}>
          <Text style={styles.waterCountText}>{waterCount} <Text style={styles.waterUnit}>Bardak</Text></Text>
          <Text style={styles.waterGoalText}>/ Hedef: {dailyWaterGoal}</Text>
        </View>
        <TouchableOpacity style={styles.waterAddButton} onPress={incrementWater}>
          <MaterialCommunityIcons name="water-plus" size={24} color="#fff" />
          <Text style={styles.waterAddButtonText}>1 Bardak Ekle</Text>
        </TouchableOpacity>
      </View>

      {/* KRONOMETRE / EGZERSİZ SÜRESİ PANELİ */}
      <View style={styles.card}>
        <Text style={styles.cardTitle}>Egzersiz Süresi Kronometresi ⏱️</Text>
        <Text style={styles.timerText}>{formatTime(timer)}</Text>
        <View style={styles.timerButtonRow}>
          <TouchableOpacity 
            style={[styles.timerButton, isTimerActive ? styles.timerButtonPause : styles.timerButtonStart]} 
            onPress={handleStartPauseTimer}
          >
            <Text style={styles.timerButtonText}>{isTimerActive ? 'Duraklat' : 'Başlat'}</Text>
          </TouchableOpacity>
          <TouchableOpacity style={[styles.timerButton, styles.timerButtonReset]} onPress={handleResetTimer}>
            <Text style={styles.timerButtonText}>Sıfırla</Text>
          </TouchableOpacity>
        </View>
      </View>

      {/* BMI VE EGZERSİZ ÖNERİSİ PANELİ */}
      <View style={styles.card}>
        <Text style={styles.cardTitle}>BMI &amp; Egzersiz Önerisi 🏃‍♂️</Text>
        <View style={styles.inputRow}>
          <TextInput 
            style={styles.input} 
            placeholder="Boy (cm)" 
            keyboardType="numeric"
            value={height}
            onChangeText={setHeight}
          />
          <TextInput 
            style={styles.input} 
            placeholder="Kilo (kg)" 
            keyboardType="numeric"
            value={weight}
            onChangeText={setWeight}
          />
        </View>
        <TouchableOpacity style={styles.calculateButton} onPress={calculateBMI}>
          <Text style={styles.calculateButtonText}>Hesapla ve Öneri Al</Text>
        </TouchableOpacity>

        {bmiResult && (
          <View style={styles.bmiResultContainer}>
            <Text style={styles.bmiResultText}>Endeksiniz: <Text style={styles.bmiHighlight}>{bmiResult}</Text> ({bmiCategory})</Text>
            <Text style={styles.exerciseRecommendationTitle}>Önerilen Aktivite:</Text>
            <Text style={styles.exerciseRecommendationText}>{exerciseRecommendation}</Text>
          </View>
        )}
      </View>

      {/* GÜNLÜK NOTLAR PANELİ */}
      <View style={styles.card}>
        <Text style={styles.cardTitle}>Günlük Notlarım 📝</Text>
        <View style={styles.noteInputRow}>
          <TextInput 
            style={[styles.input, { flex: 1, marginRight: 10 }]} 
            placeholder="Notunuzu yazın..." 
            value={currentNote}
            onChangeText={setCurrentNote}
          />
          <TouchableOpacity style={styles.noteAddButton} onPress={addNote}>
            <Text style={styles.noteAddButtonText}>Ekle</Text>
          </TouchableOpacity>
        </View>

        <View style={styles.notesList}>
          {notes.map((note) => (
            <View key={note.id} style={styles.noteItem}>
              <View style={styles.noteTextContainer}>
                <Text style={styles.noteTime}>{note.date}</Text>
                <Text style={styles.noteText}>{note.text}</Text>
              </View>
              <TouchableOpacity onPress={() => deleteNote(note.id)}>
                <MaterialCommunityIcons name="delete-outline" size={22} color="#ff5252" />
              </TouchableOpacity>
            </View>
          ))}
        </View>
      </View>

      {/* PROFİL AYARLARI MODALI */}
      <Modal
        animationType="slide"
        transparent={true}
        visible={isProfileModalVisible}
        onRequestClose={() => setIsProfileModalVisible(false)}
      >
        <View style={styles.modalOverlay}>
          <View style={styles.modalContent}>
            <Text style={styles.modalTitle}>Profil ve Hedef Ayarları</Text>
            
            <Text style={styles.label}>İsminiz:</Text>
            <TextInput 
              style={[styles.input, { width: '100%', marginBottom: 15 }]} 
              placeholder="Adınızı girin"
              value={userName}
              onChangeText={setUserName}
            />

            <Text style={styles.label}>Günlük Su Hedefi (Bardak):</Text>
            <TextInput 
              style={[styles.input, { width: '100%', marginBottom: 20 }]} 
              placeholder="Örn: 8" 
              keyboardType="numeric"
              value={dailyWaterGoal}
              onChangeText={setDailyWaterGoal}
            />

            <TouchableOpacity 
              style={styles.modalCloseButton} 
              onPress={() => setIsProfileModalVisible(false)}
            >
              <Text style={styles.modalCloseButtonText}>Kaydet ve Kapat</Text>
            </TouchableOpacity>
          </View>
        </View>
      </Modal>

    </ScrollView>
  );
}
